# Events Feature Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers-extended-cc:subagent-driven-development (recommended) or superpowers-extended-cc:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add Events as an alternative to Tasks — time-bound commitments that appear as immovable blocks on the daily schedule.

**Architecture:** New `event` package parallel to `task`. Events have their own entity, service, controller, and DTOs. TimeBlock gains an `event_id` FK (mutual exclusion with `task_id`). Schedule service materializes Event TimeBlocks at plan time. Frontend pushBlocks algorithm treats Event blocks as immovable hard boundaries.

**Tech Stack:** Java 21, Spring Boot, JPA/Hibernate, Flyway, React, TanStack Query, dnd-kit, Tailwind CSS

```sql
-- V8__create_event.sql

-- 1. Create event table
CREATE TABLE event (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID NOT NULL REFERENCES app_user(id),
    project_id      UUID NOT NULL REFERENCES project(id),
    title           VARCHAR(255) NOT NULL,
    description     TEXT,
    energy_level    VARCHAR(20),
    block_date      DATE NOT NULL,
    start_time      TIME NOT NULL,
    end_time        TIME NOT NULL,
    archived_at     TIMESTAMPTZ,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT now(),
    updated_at      TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX idx_event_user_date ON event(user_id, block_date);
CREATE INDEX idx_event_project ON event(project_id);

CREATE TRIGGER trg_event_updated_at
    BEFORE UPDATE ON event
    FOR EACH ROW EXECUTE FUNCTION update_updated_at();

-- 2. Add event_id to time_block (task_id is already nullable)
ALTER TABLE time_block
    ADD COLUMN event_id UUID REFERENCES event(id);

-- 3. Enforce exactly one of task_id or event_id is set
ALTER TABLE time_block
    ADD CONSTRAINT chk_time_block_task_or_event
    CHECK (
        (task_id IS NOT NULL AND event_id IS NULL) OR
        (task_id IS NULL AND event_id IS NOT NULL)
    );

-- 4. Add resolved_event_id to deferred_item
ALTER TABLE deferred_item
    ADD COLUMN resolved_event_id UUID REFERENCES event(id);
```

- [ ] **Step 2: Handle existing time_block rows with null task_id**

The check constraint requires exactly one non-null. If any existing rows have `task_id IS NULL`, the ALTER will fail. Check the V6 migration — `task_id` has no NOT NULL constraint, so there could be rows with null task_id from the "extend" feature. We need to handle this:

Before the CHECK constraint, add:
```sql
-- Delete orphan time_block rows that have no task (shouldn't exist in practice,
-- but the extend feature could create them if a block was extended without a task)
DELETE FROM time_block WHERE task_id IS NULL;
```

This ensures the check constraint can be applied cleanly.

```java
package com.echel.planner.backend.event;

import com.echel.planner.backend.auth.AppUser;
import com.echel.planner.backend.project.Project;
import com.echel.planner.backend.task.EnergyLevel;
import jakarta.persistence.*;

import java.time.Instant;
import java.time.LocalDate;
import java.time.LocalTime;
import java.util.UUID;

/**
 * A time-bound commitment that appears as an immovable block on the daily schedule.
 * Unlike Tasks, Events have a defined start/end time set at creation and cannot be
 * moved via drag-and-drop on the schedule grid.
 */
@Entity
@Table(name = "event")
public class Event {

    @Id
    @GeneratedValue(strategy = GenerationType.UUID)
    private UUID id;

    @ManyToOne(fetch = FetchType.LAZY, optional = false)
    @JoinColumn(name = "user_id", nullable = false)
    private AppUser user;

    @ManyToOne(fetch = FetchType.LAZY, optional = false)
    @JoinColumn(name = "project_id", nullable = false)
    private Project project;

    @Column(nullable = false, length = 255)
    private String title;

    @Column(columnDefinition = "TEXT")
    private String description;

    @Enumerated(EnumType.STRING)
    @Column(name = "energy_level", length = 20)
    private EnergyLevel energyLevel;

    @Column(name = "block_date", nullable = false)
    private LocalDate blockDate;

    @Column(name = "start_time", nullable = false)
    private LocalTime startTime;

    @Column(name = "end_time", nullable = false)
    private LocalTime endTime;

    @Column(name = "archived_at")
    private Instant archivedAt;

    @Column(name = "created_at", nullable = false, insertable = false, updatable = false)
    private Instant createdAt;

    @Column(name = "updated_at", nullable = false, insertable = false, updatable = false)
    private Instant updatedAt;

    public Event() {}

    public Event(AppUser user, Project project, String title, LocalDate blockDate,
                 LocalTime startTime, LocalTime endTime) {
        this.user = user;
        this.project = project;
        this.title = title;
        this.blockDate = blockDate;
        this.startTime = startTime;
        this.endTime = endTime;
    }

    // Getters
    public UUID getId() { return id; }
    public AppUser getUser() { return user; }
    public Project getProject() { return project; }
    public String getTitle() { return title; }
    public String getDescription() { return description; }
    public EnergyLevel getEnergyLevel() { return energyLevel; }
    public LocalDate getBlockDate() { return blockDate; }
    public LocalTime getStartTime() { return startTime; }
    public LocalTime getEndTime() { return endTime; }
    public Instant getArchivedAt() { return archivedAt; }
    public Instant getCreatedAt() { return createdAt; }
    public Instant getUpdatedAt() { return updatedAt; }

    // Setters for mutable fields
    public void setProject(Project project) { this.project = project; }
    public void setTitle(String title) { this.title = title; }
    public void setDescription(String description) { this.description = description; }
    public void setEnergyLevel(EnergyLevel energyLevel) { this.energyLevel = energyLevel; }
    public void setBlockDate(LocalDate blockDate) { this.blockDate = blockDate; }
    public void setStartTime(LocalTime startTime) { this.startTime = startTime; }
    public void setEndTime(LocalTime endTime) { this.endTime = endTime; }
    public void setArchivedAt(Instant archivedAt) { this.archivedAt = archivedAt; }
}
```

```java
package com.echel.planner.backend.event;

import org.springframework.data.jpa.repository.JpaRepository;
import org.springframework.data.jpa.repository.Query;
import org.springframework.data.repository.query.Param;

import java.time.LocalDate;
import java.util.List;
import java.util.Optional;
import java.util.UUID;

public interface EventRepository extends JpaRepository<Event, UUID> {

    Optional<Event> findByIdAndUserId(@Param("id") UUID id, @Param("userId") UUID userId);

    @Query("""
            SELECT e FROM Event e
            JOIN FETCH e.project
            WHERE e.user.id = :userId
              AND e.blockDate = :blockDate
              AND e.archivedAt IS NULL
            ORDER BY e.startTime ASC
            """)
    List<Event> findByUserIdAndBlockDate(@Param("userId") UUID userId,
                                         @Param("blockDate") LocalDate blockDate);

    @Query("""
            SELECT e FROM Event e
            WHERE e.project.id = :projectId
              AND e.user.id = :userId
              AND e.archivedAt IS NULL
            ORDER BY e.blockDate ASC, e.startTime ASC
            """)
    List<Event> findByProjectIdAndUserId(@Param("projectId") UUID projectId,
                                          @Param("userId") UUID userId);

    void deleteByProjectId(UUID projectId);

    long countByUserId(UUID userId);
    void deleteByUserId(UUID userId);
}
```

```java
package com.echel.planner.backend.event;

import com.echel.planner.backend.auth.AppUser;
import com.echel.planner.backend.event.dto.EventCreateRequest;
import com.echel.planner.backend.event.dto.EventResponse;
import com.echel.planner.backend.event.dto.EventUpdateRequest;
import com.echel.planner.backend.project.Project;
import com.echel.planner.backend.project.ProjectRepository;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.time.Instant;
import java.time.LocalDate;
import java.util.List;
import java.util.UUID;

@Service
@Transactional
public class EventService {

    private final EventRepository eventRepository;
    private final ProjectRepository projectRepository;

    public EventService(EventRepository eventRepository, ProjectRepository projectRepository) {
        this.eventRepository = eventRepository;
        this.projectRepository = projectRepository;
    }

    public EventResponse create(AppUser user, UUID projectId, EventCreateRequest request) {
        Project project = projectRepository.findByIdAndUserId(projectId, user.getId())
                .orElseThrow(() -> new EventValidationException("Project not found: " + projectId));

        if (!request.endTime().isAfter(request.startTime())) {
            throw new EventValidationException("End time must be after start time");
        }

        Event event = new Event(user, project, request.title(), request.blockDate(),
                request.startTime(), request.endTime());
        event.setDescription(request.description());
        event.setEnergyLevel(request.energyLevel());

        return EventResponse.from(eventRepository.save(event));
    }

    @Transactional(readOnly = true)
    public EventResponse get(AppUser user, UUID eventId) {
        Event event = findOwned(user, eventId);
        return EventResponse.from(event);
    }

    @Transactional(readOnly = true)
    public List<EventResponse> listByProject(AppUser user, UUID projectId) {
        return eventRepository.findByProjectIdAndUserId(projectId, user.getId())
                .stream()
                .map(EventResponse::from)
                .toList();
    }

    @Transactional(readOnly = true)
    public List<Event> findForDate(AppUser user, LocalDate date) {
        return eventRepository.findByUserIdAndBlockDate(user.getId(), date);
    }

    public EventResponse update(AppUser user, UUID eventId, EventUpdateRequest request) {
        Event event = findOwned(user, eventId);

        if (request.title() != null) event.setTitle(request.title());
        if (request.description() != null) event.setDescription(request.description());
        if (request.energyLevel() != null) event.setEnergyLevel(request.energyLevel());
        if (request.blockDate() != null) event.setBlockDate(request.blockDate());
        if (request.startTime() != null) event.setStartTime(request.startTime());
        if (request.endTime() != null) event.setEndTime(request.endTime());

        if (request.projectId() != null) {
            Project newProject = projectRepository.findByIdAndUserId(request.projectId(), user.getId())
                    .orElseThrow(() -> new EventValidationException("Project not found: " + request.projectId()));
            event.setProject(newProject);
        }

        // Validate times after all updates applied
        if (!event.getEndTime().isAfter(event.getStartTime())) {
            throw new EventValidationException("End time must be after start time");
        }

        return EventResponse.from(event);
    }

    public EventResponse archive(AppUser user, UUID eventId) {
        Event event = findOwned(user, eventId);
        event.setArchivedAt(Instant.now());
        return EventResponse.from(event);
    }

    private Event findOwned(AppUser user, UUID eventId) {
        return eventRepository.findByIdAndUserId(eventId, user.getId())
                .orElseThrow(() -> new EventNotFoundException("Event not found: " + eventId));
    }

    public static class EventNotFoundException extends RuntimeException {
        public EventNotFoundException(String message) { super(message); }
    }

    public static class EventValidationException extends RuntimeException {
        public EventValidationException(String message) { super(message); }
    }
}
```

```java
package com.echel.planner.backend.event;

import com.echel.planner.backend.event.dto.EventCreateRequest;
```

```java
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.security.core.annotation.AuthenticationPrincipal;
import org.springframework.web.bind.annotation.*;

import java.util.List;
import java.util.UUID;

@RestController
@RequestMapping("/api/v1")
public class EventController {

    private final EventService eventService;

    public EventController(EventService eventService) {
        this.eventService = eventService;
    }

    @PostMapping("/projects/{projectId}/events")
    public ResponseEntity<EventResponse> create(@AuthenticationPrincipal AppUser user,
                                                 @PathVariable UUID projectId,

    }

    @GetMapping("/projects/{projectId}/events")
    public ResponseEntity<List<EventResponse>> listByProject(@AuthenticationPrincipal AppUser user,
                                                              @PathVariable UUID projectId) {
        return ResponseEntity.ok(eventService.listByProject(user, projectId));
    }

    @GetMapping("/events/{id}")
    public ResponseEntity<EventResponse> get(@AuthenticationPrincipal AppUser user,
                                              @PathVariable UUID id) {
        return ResponseEntity.ok(eventService.get(user, id));
    }

    @PutMapping("/events/{id}")
    public ResponseEntity<EventResponse> update(@AuthenticationPrincipal AppUser user,
                                                 @PathVariable UUID id,
                                                 @Valid @RequestBody EventUpdateRequest request) {
        return ResponseEntity.ok(eventService.update(user, id, request));
    }

    @PatchMapping("/events/{id}/archive")
    public ResponseEntity<EventResponse> archive(@AuthenticationPrincipal AppUser user,
                                                  @PathVariable UUID id) {
        return ResponseEntity.ok(eventService.archive(user, id));
    }
}
```

```java
public List<TimeBlockResponse> savePlan(AppUser user, SavePlanRequest request) {
    validateBlocks(request.blocks());

    LocalDate today = LocalDate.now(ZoneId.of(user.getTimezone()));

    // Query events for this date
    List<Event> events = eventService.findForDate(user, today);

    // Validate no overlaps between submitted task blocks and events
    validateNoEventOverlaps(request.blocks(), events);

    timeBlockRepository.deleteByUserIdAndBlockDate(user.getId(), today);

    List<TimeBlock> toSave = new ArrayList<>();
    int sortIdx = 0;

    // Create event time blocks first
    for (Event event : events) {
        toSave.add(new TimeBlock(user, today, event, event.getStartTime(), event.getEndTime(), sortIdx++));
    }

    // Create task time blocks
    for (SavePlanRequest.BlockEntry entry : request.blocks()) {
        Task task = taskRepository.findByIdAndUserId(entry.taskId(), user.getId())
                .orElseThrow(() -> new ScheduleValidationException("Task not found: " + entry.taskId()));
        toSave.add(new TimeBlock(user, today, task, entry.startTime(), entry.endTime(), sortIdx++));
    }

    // Sort all blocks by start time for consistent ordering
    toSave.sort(Comparator.comparing(TimeBlock::getStartTime));
    for (int i = 0; i < toSave.size(); i++) {
        toSave.get(i).setSortOrder(i);
    }

    return timeBlockRepository.saveAll(toSave).stream()
            .map(TimeBlockResponse::from)
            .toList();
}

private void validateNoEventOverlaps(List<SavePlanRequest.BlockEntry> taskBlocks, List<Event> events) {
    for (SavePlanRequest.BlockEntry block : taskBlocks) {
        for (Event event : events) {
            if (block.startTime().isBefore(event.getEndTime()) && block.endTime().isAfter(event.getStartTime())) {
                throw new ScheduleValidationException(
                        "Task block " + block.startTime() + "–" + block.endTime() +
                        " overlaps with event '" + event.getTitle() + "' at " +
                        event.getStartTime() + "–" + event.getEndTime());
            }
        }
    }
}
```

- [ ] **Step 2: Update pushBlocks to treat event blocks as immovable**

Add an `isEvent` property to grid blocks. In `pushBlocks()`, if a block being pushed is an event, reject the operation (return null):

```javascript
export function pushBlocks(blocks, changedIndex, dayEndMinutes) {
  const result = blocks.map(b => ({ ...b }))

  for (let i = changedIndex + 1; i < result.length; i++) {
    const prev = result[i - 1]
    const curr = result[i]
    if (curr.startMinutes < prev.endMinutes) {
      // If the block that needs to move is an event, reject the entire operation
      if (curr.isEvent) {
        return null
      }
      const shift = prev.endMinutes - curr.startMinutes
      result[i] = {
        ...curr,
        startMinutes: curr.startMinutes + shift,
        endMinutes: curr.endMinutes + shift,
      }
    } else {
      break
    }
  }

  if (result.length > 0 && result[result.length - 1].endMinutes > dayEndMinutes) {
    return null
  }

  return result
}
```
